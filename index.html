4 ProGuard Configuration

To enable code obfuscation using ProGuard, configure your app-level build.gradle and create/edit the proguard-rules.pro file.

Enable Minification

build.gradle (app) — buildTypes 
buildTypes {
    release {
        minifyEnabled true
        shrinkResources true  // recommended but optional
        proguardFiles getDefaultProguardFile('proguard-android-optimize.txt'),
                      'proguard-rules.pro'
    }
}

NDK ABI Filters

Move sensitive information from hardcoded strings to NDK. Add the following configuration inside the defaultConfig block:

defaultConfig block 
ndk {
    abiFilters 'x86', 'x86_64', 'armeabi-v7a', 'arm64-v8a'
}

4.2 Signing Configuration for Consistent Trust ID

To generate a unique and consistent Trust ID across development and production builds, use the same keystore configuration for both debug and release builds.

build.gradle (:app)
android {
    signingConfigs {
        common {
            keyAlias "your_key_alias"
            keyPassword "your_key_password"
            storeFile file("path/to/your-keystore.jks")
            storePassword "your_store_password"
            v1SigningEnabled false
            v2SigningEnabled true
        }
    }

    buildTypes {
        debug {
            signingConfig signingConfigs.common
        }

        release {
            signingConfig signingConfigs.common
            minifyEnabled true
            shrinkResources true
            proguardFiles getDefaultProguardFile('proguard-android-optimize.txt'), 'proguard-rules.pro'
        }
    }
}
build.gradle.kts (:app)
android {
    signingConfigs {
        create("common") {
            keyAlias = "your_key_alias"
            keyPassword = "your_key_password"
            storeFile = file("path/to/your-keystore.jks")
            storePassword = "your_store_password"
            enableV1Signing = false
            enableV2Signing = true
        }
    }

    buildTypes {
        getByName("debug") {
            signingConfig = signingConfigs.getByName("common")
        }

        getByName("release") {
            signingConfig = signingConfigs.getByName("common")
            isMinifyEnabled = true
            isShrinkResources = true
            proguardFiles(
                getDefaultProguardFile("proguard-android-optimize.txt"),
                "proguard-rules.pro"
            )
        }
    }
}

NDKJavaKotlin

Use NDK to securely store sensitive credentials (license keys, IDs, passwords) in native C++ code, making them significantly harder to extract through reverse engineering.

Setup C++ Module

In Android Studio: Android → Project → Expand app → Expand src → right-click main → Add C++ to Module

2
Create CMakeLists.txt

3
Open the Generated .cpp File

Expand the cpp folder → open the generated .cpp file (e.g., sampleappaar.cpp).

C++ Implementation

ℹ️
Naming Convention Replace <your_package_path>_<class_name> with your package path using underscores (e.g., com_example_app_MyClass) for JNI method names. Replace <your/package/path> with slashes for FindClass calls (e.g., com/example/app).
sampleappaar.cpp 
#include <jni.h>
#include <string>

// Add all given credentials from the release document
std::string packageName       = "";  // keep package name here
std::string appName           = "";  // keep app name here
std::string channelLicenseKey = "";  // keep channel licence key here
std::string clientId          = "";  // keep client id here
std::string channelId         = "";  // keep channel id here
std::string password          = "";  // keep password here

extern "C" JNIEXPORT jstring JNICALL
Java_<your_package_path>_<class_name>_appPackageName(JNIEnv *env, jobject) {
    return env->NewStringUTF(packageName.c_str());
}

extern "C" JNIEXPORT jstring JNICALL
Java_<your_package_path>_<class_name>_appName(JNIEnv *env, jobject) {
    return env->NewStringUTF(appName.c_str());
}

extern "C" JNIEXPORT jstring JNICALL
Java_<your_package_path>_<class_name>_channelLicenseKey(JNIEnv *env, jobject) {
    return env->NewStringUTF(channelLicenseKey.c_str());
}

extern "C" JNIEXPORT jstring JNICALL
Java_<your_package_path>_<class_name>_clientId(JNIEnv *env, jobject) {
    return env->NewStringUTF(clientId.c_str());
}

extern "C" JNIEXPORT jstring JNICALL
Java_<your_package_path>_<class_name>_channelId(JNIEnv *env, jobject) {
    return env->NewStringUTF(channelId.c_str());

6 Advanced Reverse Engineering Control

This section adds an additional layer of protection by embedding load-check logic in native C++ code that is called from the launcher activity.

1. Add NDK Methods in .cpp File (Set Launcher Path)

ℹ️
Naming Replacements
  • Replace <your_package_path>_<LauncherActivity> with your specific class package path
  • Replace <your_package_path> using format com_example_app for Java method naming
  • Replace <your/package/path> using format com/example/app for FindClass
sampleappaar.cpp — Reverse Engineering Control 
extern "C"
JNIEXPORT jboolean JNICALL
Java_<your_package_path>_<LauncherActivity>_getLoadCheck(JNIEnv *env, jobject thiz) {
    jclass scanCoreClass = env->FindClass("klo/yui/bnm/NativeInteractor");
    jmethodID initScanCore = env->GetMethodID(scanCoreClass, "<init>", "()V");
    jobject scanCoreObject = env->NewObject(scanCoreClass, initScanCore);
    jmethodID getMethodID = env->GetMethodID(scanCoreClass, "getIslibLoadApp", "()Z");
    jboolean result = env->CallBooleanMethod(scanCoreObject, getMethodID);
    return result;
}

extern "C"
JNIEXPORT jstring JNICALL
Java_<your_package_path>_<LauncherActivity>_close(JNIEnv *env, jobject thiz) {
    abort();
}

extern "C" void Java_<your_package_path>_<LauncherActivity>_getLoggingStatus(
        JNIEnv* env, jobject obj) {
    jclass clazz = env->FindClass("your_package_path/Launcher_activity");
    jmethodID status = env->GetMethodID(clazz, "logStatus", "()V");
    env->CallVoidMethod(obj, status);
}

Section 07

7 Advanced Threat Signal (ATS) SDK

The Advanced Threat Signal (ATS) SDK for Android collects device integrity signals, packages them securely, and encrypts them end-to-end before transmission to your backend. It is designed to detect tampering, emulation, and bypass attempts at the device level.

📖
Backend Plugin Integration The backend Java plugin handles the server-side processing of signals. It decrypts the encrypted signal, validates integrity parameters, returns an ALLOW / BLOCK verdict, and integrates with your fraud/risk engine.

Refer section 7.1 — Java Backend Plugin Integration.

The FDAnalytics SDK (ai.protectt.sdk.analytics) collects device threat signals, packages them into a compact payload, and encrypts that payload end-to-end. It is the client-side counterpart to the backend Java plugin.

Requirements

ℹ️
Zero External Dependencies The SDK uses only Android platform APIs and the JDK's built-in javax.crypto / java.security stack — no third-party libraries required.
ItemMinimum
Android minSdk24 (Android 7.0)
compileSdk34+
Kotlin1.9+
Java source compatibilityJava 17
Native ABIarmeabi-v7a, arm64-v8a, x86_64

Registering Threat Events

Use AppProtecttInteractor to listen for threat events and forward them to the SDK.

Java — Event Registration 
FDAnalytics sdk = FDAnalytics.getInstance();

AppProtecttInteractor.recevieEventCallback(
    new AppProtecttInteractorContract.EventNotified() {

        @Override
        public void onEvents(Response response) {
            sdk.addOnEvent(response.vulnerabilityCode);
        }

        @Override
        public void initEvents(int response) {
            sdk.addInitEvent(response);
        }
    }
);
Kotlin — Event Registration 
val sdk = FDAnalytics.getInstance()

AppProtecttInteractor.recevieEventCallback(
    object : AppProtecttInteractorContract.EventNotified {
        override fun onEvents(response: Response) {
            sdk.addOnEvent(response.vulnerabilityCode)
        }

        override fun initEvents(response: Int) {
            sdk.addInitEvent(response)
        }
    }
)

Event Methods

MethodPurpose
addOnEvent(int ruleId)Records a vulnerability event
addInitEvent(int ruleId)Records an initialization event

Generating the Signal

Call getSignal() to produce an encrypted signal string, then transmit it to your server.

Java — Generate & Send Signal 
try {
    FDAnalytics sdk = FDAnalytics.getInstance();

    String signal = sdk.getSignal(
            "<your-session-id>",
            "<your-device-id>",
            "<unix-timestamp-ms>"
    );

    sendToServer(signal);

} catch (Exception e) {
    Log.e("FDAnalyticsError", "Error generating encrypted signal", e);
}
Kotlin — Generate & Send Signal 
try {
    val sdk = FDAnalytics.getInstance()

    val signal = sdk.getSignal(
        "<your-session-id>",
        "<your-device-id>",
        "<unix-timestamp-ms>"
    )

    sendToServer(signal)
} catch (e: Exception) {
    Log.e("FDAnalyticsError", "Error generating encrypted signal", e)
}

API Transmission & Encoding

⚠️
Critical — Base64 Characters The encrypted signal is standard Base64 and may contain +, /, and = characters. Avoid sending the signal as a raw URL query parameter, as special characters will be mangled by URL encoding.

Recommended approach — send the signal via a JSON POST body:

JSON — POST Body 
{
  "signal": "<encrypted-signal>"
}

Signal Plaintext Format

The decrypted signal follows this structure:

Signal Format 
<sessionId>#<protecttUUID>#<deviceId>#<threatParams>[#<sessionTimeStamp>]

Integration Flow

Follow these steps in order for a complete integration:

  1. Initialize the SDK with the server public key.
  2. Register threat event callbacks.
  3. Generate an encrypted signal per session.
  4. Send the signal via JSON POST.
  5. Backend decrypts and validates the signal.
  6. Enforce the verdict (ALLOW / BLOCK) in your app.

Security Best Practices

🔒
Follow these guidelines to maintain the integrity of the ATS integration:
  • Never expose the private key in the app binary or source code.
  • Rotate server keys periodically to limit exposure from key compromise.
  • Enforce HTTPS with TLS 1.2 or higher on all endpoints.
  • Validate timestamp freshness server-side to prevent replay attacks.

Follow these steps when uploading your app to the Google Play Store for the first time with AppProtectt integrated.

1
Add Protectt.ai as Testers

Add the following email addresses as testers in the Internal Testing Channel on Google Play Console:

  • protectt.ai@gmail.com
  • protectt.apptesting@gmail.com

2

Upload the production build to the Beta Channel (Internal Testing) on the Google Play Store.

During this upload, enable the following LOG in your Application class to extract the TrustID hash value:

Application.java / .kt 
Log.i("TrustID", AppProtecttInteractor.Companion.getTrust(this));
3
Trust Value Extraction

After the build is available on the Beta Channel, the trust value will be extracted from the log by the Protectt.ai team. This trust value is used for further configuration of the AppProtectt Security solution.

4
File Replacement

Protectt.ai will share an updated featureprod.png file. Replace it in res/raw/ of your Android project. Once updated, remove the TrustID LOG used in Step 2.

5
Upload the Updated Build

Upload the updated build (with featureprod.png replaced and LOG removed) to the Internal Beta Channel on Google Play Store. Keep it accessible briefly to allow the Protectt.ai team to run a sanity test.

Section 09

9 Migrating from AAR File to Maven

If your project previously integrated AppProtectt via a local .aar file, follow the steps outlined in the Dependencies section (Section 2) to switch to the Maven repository approach.

Key changes to make when migrating:

  • Remove any local .aar files from your project's libs folder
  • Remove any flatDir repository entries referencing local libs
  • Add the Protectt.ai Maven repository URL and credentials as described in Section 2
  • Replace local file dependencies with the Maven coordinate: implementation "ai.protectt.sdk:appprotectt:X.X"
  • Add Maven credentials to local.properties (or directly in the Gradle file for CI/CD)
  • Sync Gradle and verify the build succeeds
✅
Migration Complete Once migrated to Maven, you will automatically receive SDK updates by bumping the version number in your dependency declaration, without needing to manually download and replace .aar files.

Section 7 — Backend Plugin

7.6 Integration Flow

1. Frontend / Client

  • Sends request with clientSessionID and signalToken
  • signalToken contains encrypted data: (SessionId#Pbid#DeviceId#ThreatParams#SessionTimestamp)

2. Plugin Validation

The plugin checks required fields (clientSessionID, signalToken, clientId, chnId), decodes the signalToken to extract pbid, deviceId, and threatParams, then returns a preliminary decision:

  • BLOCK if the signal is invalid
  • ALLOW if the signal is valid

3. Session Verification

Once plugin.validate(req) returns a successful response, you must verify:

⚠️
Session Match — Ensure response.get("sessionId") matches the session stored in your backend.

Time TTL Verification — Compare the timestamp from the response with your server's current timestamp. Reject the signal if the difference exceeds 5 minutes (300,000 ms) to prevent replay attacks.

4. Plugin Forwarding

The plugin forwards the decoded signal to the Protectt server endpoint /threat/validateplugin. The server performs additional checks including device blacklist status, then returns ALLOW or BLOCK with a reason.

5. Plugin Final Decision

The plugin combines local validation and Protectt server response, blocks and blacklists the device if the server returns BLOCK, and returns the final JSON (action, reason, code, pbid, sessionId, deviceId, timestamp) to the frontend.

11 UPI Fraud Prevention Guidelines

This section outlines a layered defence strategy for UPI-enabled Android applications. Each layer independently reduces fraud surface area; combined, they provide defence-in-depth against the most common attack vectors targeting mobile payment flows.

🔒
Mandatory for Financial Apps All four controls below should be treated as non-optional for any application that initiates, authorises, or displays UPI payment flows. Omitting any single layer substantially increases the risk of successful fraud.
Section 11.1

12.1 Enable Auto Play Protection

Ensure your app is protected against tampering, repackaging, and distribution from untrusted sources. Apps distributed through side-loading bypass Google Play's automated malware scanning and integrity verification, leaving users exposed to trojans and credential-harvesting overlays.

Key Actions

  • Enable Auto Play Protection from Google Play Console
  • Restrict app usage if installed from unknown or unofficial sources
  • Detect and block sideloaded or modified APKs at runtime

Steps to Enable

1

Navigate to play.google.com/console and select your application.

2
Open App Integrity Settings

Go to Setup → App Integrity (or the App Signing & Protection section, depending on your console version).

3
Enable Automatic Protection & Play App Signing

Toggle on Automatic Protection and Play App Signing as applicable for your release track.

4
Configure Enforcement Rules

Set enforcement rules to block the app if not installed via the Play Store, and to restrict usage on tampered or repackaged builds.

Benefits

Anti-Repackaging
Prevents redistribution

Section 11.2

12.2 Implement Play Integrity API

The Google Play Integrity API provides a cryptographically signed integrity verdict that your backend can verify before completing any sensitive operation. Unlike client-side checks, which can be bypassed by a motivated attacker, backend verification of the integrity token is significantly harder to circumvent.

Purpose

  • Detect rooted or compromised devices before allowing transactions
  • Ensure the app binary is the one published to the Play Store
  • Prevent usage of tampered, cloned, or instrumented apps

Integration Flow

Play Integrity — High-Level Flow 
// 1. Request an integrity token from Google
IntegrityTokenRequest request = IntegrityTokenRequest.builder()
        .setNonce(generateNonce())          // server-generated nonce
        .build();

IntegrityManager integrityManager =
        IntegrityManagerFactory.create(applicationContext);

integrityManager.requestIntegrityToken(request)
        .addOnSuccessListener(response -> {
            String token = response.token();
            // 2. Send token to your backend for verification
            sendToBackend(token);
        })
        .addOnFailureListener(e -> {
            // Handle failure — treat as high-risk
        });

Best Practices

  • Always validate the integrity token on the backend — never trust client-side results alone
  • Enforce checks at login, OTP verification, and every transaction initiation
  • Use a fresh server-generated nonce per request to prevent token replay
  • Block or flag devices with a MEETS_BASIC_INTEGRITY = false verdict
  • Log and alert on unusually high volumes of integrity failures from a single device ID
⚠️
Nonce Replay Prevention Generate the nonce server-side per request and verify it has not been used before. Reusing nonces allows an attacker to replay a previously obtained valid token against multiple transactions.
Section 11.3

12.3 Enable Advanced Threat Signaling (ATS)

Advanced Threat Signaling provides continuous, real-time device integrity monitoring throughout the user session — not just at app launch. Where Play Integrity gives a point-in-time verdict, ATS tracks threat events as they evolve (e.g., a root tool being activated mid-session) and surfaces them to your backend for dynamic risk scoring.

ℹ️
Implementation Reference ATS integration is fully documented in Section 7 — Advanced Threat Signal (ATS) SDK (including the Java Backend Plugin sub-section 7.1–7.7) above. Refer to those sections for the complete SDK setup, event registration, signal generation, and server-side validation flow.

Coverage Checkpoints

Ensure ATS signals are collected and validated at all critical touchpoints:

TouchpointATS ActionRecommended Verdict Enforcement
App Launch / OnboardingCollect initial signalBlock if BLOCK verdict received
Login / AuthenticationRefresh signal per sessionDeny login on anomaly detection
OTP EntryValidate device continuityFlag session for step-up auth
UPI Transaction InitiationGenerate fresh encrypted signalHard block on BLOCK verdict
High-value TransferRe-validate before authorisationRequire additional factor + block on BLOCK

Why ATS Complements Play Integrity

  • Session-level coverage — Play Integrity is checked once; ATS monitors continuously
  • Custom threat rules — ATS exposes granular rule IDs (e.g., overlay detected, accessibility abuse) for fine-grained risk logic
  • Backend-verified encryption — signals are end-to-end encrypted and validated server-side, preventing client tampering
  • Replay protection — signal includes a session timestamp verified by the backend plugin
Section 11.4

12.4 Encrypt All Requests and Responses

Transport-layer security alone is not sufficient for financial applications. Payload-level encryption ensures that even if TLS is terminated by a proxy or the device's certificate trust store has been manipulated, sensitive data remains protected.

Key Measures

  • Enforce HTTPS with TLS 1.2 minimum (prefer TLS 1.3) on all API endpoints — reject downgrades
  • Implement Certificate Pinning to prevent MITM via rogue CAs or user-installed certificates
  • Apply payload-level encryption (e.g., AES-256-GCM) on sensitive request/response bodies — device signals, account numbers, UPI VPAs
  • Use ephemeral session keys derived per transaction to limit blast radius of key compromise
  • Never store sensitive data (credentials, tokens, UPI PIN) in SharedPreferences, local databases, or logs in plaintext
  • Apply Android Keystore for storing cryptographic keys — keys never leave the secure hardware enclave

Certificate Pinning — Quick Reference

res/xml/network_security_config.xml 
<?xml version="1.0" encoding="utf-8"?>
<network-security-config>
    <domain-config cleartextTrafficPermitted="false">
        <domain includeSubdomains="true">api.yourbank.com</domain>
        <pin-set expiration="2026-01-01">
            <!-- Primary public key pin (SHA-256, Base64) -->
            <pin digest="SHA-256">AAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAA=</pin>
            <!-- Backup pin (rotate before primary expires) -->
            <pin digest="SHA-256">BBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBB=</pin>
        </pin-set>
    </domain-config>
</network-security-config>
ℹ️
Reference this config in AndroidManifest.xml via android:networkSecurityConfig="@xml/network_security_config" on the <application> tag. Always maintain a backup pin and rotate before the expiration date to avoid locking out users.

Benefits

ThreatMitigation

🔒
Defence-in-Depth Summary No single control eliminates fraud risk entirely. The combination of Auto Play Protection (distribution integrity) + Play Integrity API (device & app attestation) + Advanced Threat Signaling (continuous session monitoring) + End-to-End Encryption (data-in-transit protection) provides layered coverage that raises the cost of attack to a level that deters all but the most sophisticated threat actors.

public class MyApp extends Application {

    static {
        System.loadLibrary("sampleappaar"); // match CMakeLists.txt project name
    }

    public native String appPackageName();
    public native String appName();
    public native String channelLicenseKey();
    public native String clientId();
    public native String channelId();
    public native String password();

    @Override
    public void onCreate() {
        super.onCreate();

        // ── Register callbacks before anything else ──
        AppProtecttInteractor.Companion.registerCallbackActivities(this);

        // ── [DEBUG ONLY] Remove this log before production release ──
        Log.i("TrustID", AppProtecttInteractor.Companion.getTrust(this));

        // ── Populate client info from NDK ──
        ClientInfo clientInfo = new ClientInfo();
        clientInfo.setPackageName(appPackageName());
        clientInfo.setAppName(appName());
        clientInfo.setChannelLicenseKey(channelLicenseKey());
        clientInfo.setClientId(Integer.parseInt(clientId()));
        clientInfo.setChannelId(Integer.parseInt(channelId()));
        clientInfo.setPassword(password());
        clientInfo.setMainAppVersionCode(String.valueOf(BuildConfig.VERSION_CODE));
        clientInfo.setAppVersionName(BuildConfig.VERSION_NAME);
        AppProtecttInteractor.Companion.setClientInfo(clientInfo);

        // ── Initialise SDK ──
        new AppProtecttInteractor(this).initAppProtectt(
            "SplashActivity",               // launcher activity simple name
            R.layout.alert_layout,          // pass the layout file downloaded from the UI resource package
            R.drawable.ic_app_logo,         // your app logo drawable
            0,                              // style (R.style.AlertDialogCustom for Flutter)
            BuildConfig.BUILD_TYPE,
            2,                              // 1 = AlertDialog, 2 = BottomSheet
            BuildConfig.APP_FLAVOR,
            BuildConfig.APP_BUILD_TIME
        );

        AppProtecttInteractor.Companion.updateCustRefIdAPI("YOUR_DEVICE_ID");
    }
}

        }

    private external fun appName(): String
    private external fun channelLicenseKey(): String
    private external fun clientId(): String
    private external fun channelId(): String
    private external fun password(): String

        super.onCreate()

        // ── Register callbacks before anything else ──
